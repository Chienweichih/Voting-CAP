\begin{frame}{POV - Proof of Violation}
	定義以下三個 Tuples：
    ~\\
    ~\\
	\begin{itemize}  
        \item Properties
	        \begin{itemize}
            	\item Data Integrity
                \item Write Serializability
                \item Read Freshness
            \end{itemize}
        \item Attestations
	        \begin{itemize}
            	\item 在 User 和 Service Provider 之間的訊息加上簽章，\\ 藉由此密碼學的證據能讓雙方不可否認自己做過的事
            \end{itemize}
        \item Auditing
            \begin{itemize}
            	\item 利用收集的 Attestations 來證明是否違反 Properties
            \end{itemize}
    \end{itemize}
\end{frame}
